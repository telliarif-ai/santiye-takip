Supabase Bağlantısı
SQL — Tabloları Oluştur

Supabase SQL Editor'e gidin ve aşağıdaki kodu çalıştırın:

CREATE TABLE santiyeler (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  ad TEXT NOT NULL, sehir TEXT, adres TEXT,
  ilerleme_yuzde INT DEFAULT 0,
  durum TEXT DEFAULT 'normal',
  baslangic DATE, bitis DATE,
  olusturma TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE kullanicilar (
  id UUID PRIMARY KEY REFERENCES auth.users(id),
  ad_soyad TEXT, rol TEXT DEFAULT 'saha',
  santiye_id UUID REFERENCES santiyeler(id),
  olusturma TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE gunluk_raporlar (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  santiye_id UUID REFERENCES santiyeler(id),
  kullanici_id UUID REFERENCES kullanicilar(id),
  tarih DATE DEFAULT CURRENT_DATE,
  yapilan_isler TEXT, yarin_plan TEXT,
  hava TEXT, ilerleme_yuzde INT, sorun_tipi TEXT,
  olusturma TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE personel_devam (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  santiye_id UUID REFERENCES santiyeler(id),
  kullanici_id UUID REFERENCES kullanicilar(id),
  tarih DATE DEFAULT CURRENT_DATE,
  vardiya TEXT, durum TEXT,
  fazla_mesai_saat INT DEFAULT 0
);
CREATE TABLE malzeme_giris (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  santiye_id UUID REFERENCES santiyeler(id),
  kullanici_id UUID REFERENCES kullanicilar(id),
  malzeme_adi TEXT, birim TEXT,
  kullanilan_miktar FLOAT, kalan_stok FLOAT,
  min_esik FLOAT, tarih DATE DEFAULT CURRENT_DATE, not TEXT
);
CREATE TABLE ig_olaylari (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  santiye_id UUID REFERENCES santiyeler(id),
  kullanici_id UUID REFERENCES kullanicilar(id),
  olay_tipi TEXT, aciklama TEXT, olay_yeri TEXT,
  mudahale TEXT, durum TEXT DEFAULT 'acik',
  olay_zamani TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE fotograflar (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  santiye_id UUID REFERENCES santiyeler(id),
  kullanici_id UUID REFERENCES kullanicilar(id),
  kategori TEXT, dosya_url TEXT, aciklama TEXT,
  yukleme_zamani TIMESTAMPTZ DEFAULT NOW()
);
